\begin{denotation}
     \item[\textbf{\large 符号约定}] \mbox{}
     \vspace{0.5em}
     \item[$\mathcal{I}$] 惯性坐标系 $O\text{-}xyz$，坐标轴朝向为北东地（NED）
     \item[$\mathcal{B}$] 机体坐标系 $O_b\text{-}x_by_bz_b$，其原点 $O_b$ 位于机体浮心（CB）
     \item[] 
     \item[\textbf{\large 缩略语说明}] \mbox{}
     \vspace{0.5em}
     \item[PID] 比例—积分—微分（Proportion-Integration-Differentiation）
     \item[MHE] 滚动时域估计（Moving Horizon Estimation）
     \item[MPC] 模型预测控制（Model Predictive Control） 
     \item[IMU] 惯性测量单元（Inertial Measurement Unit）
     \item[ESC] 电子调速器（Electronic Speed Control）
     \item[SoC] 片上系统（System on a Chip）
\end{denotation}
